CREATE TABLE events (
  id UUID PRIMARY KEY,
  user_id UUID,
  session_id TEXT,
  event_name TEXT,
  event_data JSONB,
  timestamp TIMESTAMP
);

-- Aggregated metrics
CREATE TABLE metrics (
  date DATE,
  metric_name TEXT,
  value NUMERIC,
  dimensions JSONB
);
